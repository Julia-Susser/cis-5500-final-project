**NYC Taxi Data:**


SELECT COUNT(*) FROM taxi;



SELECT COUNT(*) FROM information_schema.columns WHERE table_name = 'taxi';



SELECT COUNT(DISTINCT vendor_id) FROM taxi;



SELECT ROUND(AVG(surcharge), 2) FROM taxi;



SELECT ROUND(AVG(mta_tax), 2) FROM taxi;



SELECT COUNT(*) 
FROM taxi t
JOIN zone pu ON t.pu_location_id = pu.location_id
JOIN zone do ON t.do_location_id = do.location_id
WHERE pu.borough = do.borough;



SELECT ROUND(AVG(fare_amount), 2) FROM taxi;

**NYC Collision Data:**


SELECT COUNT(*) FROM collision;



SELECT COUNT(*) FROM information_schema.columns WHERE table_name = 'collision';



SELECT MIN(crash_date) FROM collision;



SELECT MAX(crash_date) FROM collision;



SELECT COUNT(*) FROM collision WHERE number_of_persons_injured > 0;



SELECT COUNT(*) FROM collision WHERE number_of_persons_killed > 0;



**NYC Geometry Data:**


SELECT COUNT(*) FROM zone;



SELECT COUNT(*) FROM information_schema.columns WHERE table_name = 'zone';



SELECT COUNT(DISTINCT location_id) FROM zone;



SELECT COUNT(DISTINCT borough) FROM zone;



SELECT borough, COUNT(*) AS zone_count
FROM zone
GROUP BY borough
ORDER BY zone_count DESC
LIMIT 1;
